A company tracks employee visits, allowing only one entry per employee per day, although a loophole exists where employees can enter multiple times if they use a different email ID for each entry.

Given this input data, determine:

  • Name of each person
  • Total number of times they visited
  • The floor they visited the most frequently
  • A comma-separated list of distinct resources they used

The output should include: Name, Total Visits, Most Visited Floor, Used Resources.

WITH floor_rank AS (
    -- Find how many times each person visited each floor
    SELECT 
        name,
        floor,
        COUNT(*) AS floor_visit_count,
        RANK() OVER (
            PARTITION BY name 
            ORDER BY COUNT(*) DESC
        ) AS rn
    FROM resources_data
    GROUP BY name, floor
),

distinct_resources AS (
    -- Get distinct resources used by each person
    SELECT DISTINCT 
        name, 
        resources 
    FROM resources_data
),

resource_list AS (
    -- Convert resource list into comma-separated format
    SELECT 
        name, 
        STRING_AGG(resources, ',') AS used_resources
    FROM distinct_resources
    GROUP BY name
),

total_visits AS (
    -- Count total visits of each employee (even with loophole multiple emails)
    SELECT 
        name,
        COUNT(*) AS total_visits
    FROM resources_data
    GROUP BY name
)

-- Final Output
SELECT 
    FR.name,
    FR.floor AS most_visited_floor,
    TV.total_visits,
    RL.used_resources
FROM floor_rank FR
JOIN total_visits TV ON FR.name = TV.name
JOIN resource_list RL ON FR.name = RL.name
WHERE FR.rn = 1;  -- Only the most visited floor
